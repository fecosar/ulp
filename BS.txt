CREATE DATABASE UNIVERSIDAD;
USE UNIVERSIDAD;
CREATE TABLE ALUMNO (
	ID INT(11) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    NOMBRE VARCHAR(30),
    FECNAC DATETIME,
    ACTIVO TINYINT(1)
);
CREATE TABLE MATERIA (
	ID INT(11) NOT NULL PRIMARY KEY AUTO_INCREMENT,
  	NOMBRE VARCHAR(30)  
);
CREATE TABLE CURSADA (
	ID INT(11) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    IDALUMNO INT(11),
    IDMATERIA INT(11),
    NOTA INT(11),
    FOREIGN KEY (IDALUMNO) REFERENCES ALUMNO (ID),
    FOREIGN KEY (IDMATERIA) REFERENCES MATERIA (ID)
);
